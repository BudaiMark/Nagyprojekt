45       * {@code startAction()}Egy gomb megnyomásának hatására lefut a metódus amiben betöltjük az új fxml-t.
46       * @param actionEvent A gombhoz tartozó esemény.
47       * @throws IOException A lefutása közben adódó Input-Output kivétel.
48       */
49  
50      public void restartAction(ActionEvent actionEvent) throws IOException {
51  
52          FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("/fxml/gamemode.fxml"));
53          Parent root = fxmlLoader.load();
54          Stage stage = (Stage) ((Node) actionEvent.getSource()).getScene().getWindow();
55          stage.setScene(new Scene(root));
56          stage.show();
57          Logger.info("Betöltődik a játékmódválasztó felület.");
58  
59  
60      }
