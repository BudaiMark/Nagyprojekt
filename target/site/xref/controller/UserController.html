16  public class UserController {
17  
18      private static final String STANDARD= "-fx-border-color: black;";
19  
20      @FXML
21      private Button continueButton;
22  
23      @FXML
24      private TextField userName;
25  
26      @FXML
27      private Label errorLabel;
28  
29  
30      public void continueAction(ActionEvent actionEvent) throws IOException {
31          if (!userName.getText().isEmpty()){
32              FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("/fxml/gamemode.fxml"));
33              Parent root = fxmlLoader.load();
34              GameModeController controller = fxmlLoader.getController();
35              controller.setUserName(userName.getText());
36              Stage stage = (Stage) ((Node) actionEvent.getSource()).getScene().getWindow();
37              stage.setScene(new Scene(root));
38              stage.show();
39              Logger.info("Az első kérdés betöltődik.");
40  
41          }else{
42  
43              errorLabel.setText("Játékos neve üres");
44              Logger.warn("Játékos neve üres");
45  
46          }
47      }
48      public void initialize(){
49  
50          continueButton.setStyle(STANDARD);
51          Logger.info("Betöltődik az LaunchController initialize() függvényben szereplő Button és Textfield vizuális beállítása.");
52  
53  
54      }
55  
56  }
