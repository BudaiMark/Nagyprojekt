1   package controller;

3   import javafx.application.Platform;
4   import javafx.event.ActionEvent;
5   import javafx.fxml.FXML;
6   import javafx.fxml.FXMLLoader;
7   import javafx.scene.Node;
8   import javafx.scene.Parent;
9   import javafx.scene.Scene;
10  import javafx.scene.control.Button;
11  import javafx.stage.Stage;
12  import org.tinylog.Logger;
13  
14  import java.io.IOException;
15  /**
16   *
17   * {@code LaunchController}A screen-en megjelenő vizuális eszközökkel történő műveletek, események itt történnek kezelésre(eventek), továbbá
18   * itt tudunk módosítani a képernyő tartalmán.
19   *
20   */
21  
22  
23  public class LaunchController {
24      /**
25       * {@code STANDARDRD}Alapvető érték definiálása a gombok számára
26       */
27      private static final String STANDARD= "-fx-border-color: black;";
28  
29      /**
30       * {@code startButton} A képernyőn látható vizuális eszköz példányosítása, az fxml-ből érjük el id alapján
31       */
32  
33  
34      @FXML
35      private Button startButton;
36  
37      @FXML
38      private Button quitButton;
39  
40  
41      /**
42       * {@code startAction()}Egy gomb megnyomásának hatására lefut a metódus amiben betöltjük az új fxml-t.
43       * @param actionEvent A gombhoz tartozó esemény.
44       * @throws IOException A lefutása közben adódó Input-Output kivétel.
45       */
46  
47      public void startAction(ActionEvent actionEvent) throws IOException {
48  
49              FXMLLoader fxmlLoader = new FXMLLoader(getClass().getResource("/fxml/user.fxml"));
50              Parent root = fxmlLoader.load();
51              Stage stage = (Stage) ((Node) actionEvent.getSource()).getScene().getWindow();
52              stage.setScene(new Scene(root));
53              stage.show();
54              Logger.info("Betöltődik a konvertáló felület.");
55  
56  
57      }
58  
59      public void quitGame(ActionEvent actionEvent) throws IOException{
60          Logger.info("Bezáródik a Snake játék.");
61          Platform.exit();
62  
63      }
64  
65      /**
66       * {@code initialize()}Ez a metódus fut le először, a konstruktorhoz nagyon hasonló, azonban innen elérjük az fxml fájl tagjait.
67       */
68      @FXML
69      public void initialize(){
70  
71          startButton.setStyle(STANDARD);
72          quitButton.setStyle(STANDARD);
73          Logger.info("Betöltődik az LaunchController initialize() függvényben szereplő Button és Textfield vizuális beállítása.");
74  
75  
76      }
77  
78  }
